如何使用ThreeJS来绘制线条

画线

Three.js 中，创建线的类主要有Line(连续线，头尾不相连)，LineSegments(线段，两个点间无需相连)，LineLoop(连续线，头尾相连); 与线相关的材质主要有 LineBasicMaterial(实线) 和 LineDashedMaterial(虚线)

因为线是可以由两个点之间连接起来的，所以材质可以使用PointsMaterial，再调用线类进行链接即可。几何体使用基础缓冲几何体 bufferGeometry

思路

方法一

  1. 定义基础几何体
  2. 定义点材质，确定点位置
  3. 以几何体和材质做为参数创建线的类
  4. 把类添加到场景中

方法二

  1. 定义基础几何体
  2. 定义线材质
  3. 定义线类，通过线类的position来规定渲染位置
  4. 把类添加到场景中

相关API

线类(Line)

连续线，各个点之前相连，首尾不连接

参数：

parameters values marks
geometry (可选)Geometry 几何体实例
material (可选)Material 材质

属性：

公用属性可以参考 基类Ojbect3D

Properties values marks
isLine Boolean
default: true		 检查当前类或派生类是否为行
.geometry Geometry 定义几何体
.material Material 材质

方法：

公用方法可以参考 基类Ojbect3D

Methods params return marks
.computeLineDistances() Line 重新计算顶点数组，将顶点按照索引顺序依次连线产生的几何线段的长度(绘制虚线必需要调用此方法)
.raycast raycaster : Raycaster
intersects : Array		 null 获取光线投射与当前线的交点
.clone() null Line 返回当前点对象

线-基础材质(LineBasicMaterial)

绘制线框的基础材质

示例

var material = new THREE.LineBasicMaterial( {
  color: 0xffffff,
  linewidth: 1,
  linecap: 'round', //ignored by WebGLRenderer
  linejoin:  'round' //ignored by WebGLRenderer
} );

参数：
PointsMaterial( parameters : Object )

参数是一个Object，key-value 都是基于基类Material，可参考Material文档
唯一例外的是颜色属性 color, 它使用 16进制字符串传入，默认为 0xffffff。

属性：

公用属性可以参考 基类Material

Properties values marks
.color Color
default: 0xffffff		 定义材质的颜色（16进制字符串）
.isLineBasicMaterial Boolean
default: true		 用于检查该类或者派生类是否为一个线材质。此类不应该被更改。
.lights Boolean
default: true		 材质是否受光线影响
.linewidth Float
default: 1		 控制线宽。由于webgl的呈现限制，这里无论设置值多少，linewidth 始终为 1
.linecap String:”butt、round、square”
default: round		 线头尾的样式。这个属性用于2D Canvas 的 linecap 属性。不适用于 webGL中
.linejoin String:”round、bevel、miter”
default: round		 定义线的连接处样式。这个属性用于2D Canvas 的 linecap 属性。不适用于 webGL中

方法：

公用方法可以参考 基类Material

线-虚线材质(LineDashedMaterial)

可以使用虚线来绘制线框的材质

示例

var material = new THREE.LineDashedMaterial( {
  color: 0xffffff,
  linewidth: 1,
  scale: 1,
  dashSize: 3,
  gapSize: 1,
} );

参数：
PointsMaterial( parameters : Object )

参数是一个Object，key-value 都是基于基类Material，可参考Material文档
唯一例外的是颜色属性 color, 它使用 16进制字符串传入，默认为 0xffffff。

属性：

公用属性可以参考 基类Material

Properties values marks
.color Color
default: 0xffffff		 定义材质的颜色（16进制字符串）
.dashSize number
default: 3		 虚线的大小
.gapSize number
default: 1		 虚线间距大小
.isLineDashedMaterial Boolean
default: true		 用于检查该类或者派生类是否为一个虚线材质。此属性值不应该被更改。
.lights Boolean
default: false		 材质是否受光照影响
.linewidth Float
default: 1		 控制线宽。由于webgl的呈现限制，这里无论设置值多少，linewidth 始终为 1
.scale number
default: 1		 虚线部分比例

import React from 'react'
import * as THREE from 'three'

class ThreeDrawBlockLine extends React.Component{
  constructor(props){
    super(props)

    this.camera = null
    this.scene = null
    this.renderer = null
  }

  render(){
    return(
      <div ref="canvasBox" className="canvas-box"></div>
    )
  }


  componentDidMount(){
    this.run()
  }
  initRenderer(){
    this.renderer = new THREE.WebGLRenderer()
    this.renderer.setSize(window.innerWidth, window.innerHeight)
    this.renderer.setClearColor(0xffffff)
    this.refs.canvasBox.appendChild(this.renderer.domElement)
  }
  initCamera(){
    this.camera = new THREE.OrthographicCamera(window.innerWidth/-2, window.innerWidth/2, window.innerHeight/2, window.innerHeight/-2, 1, 1000)
    this.camera.position.z = 1000
    this.camera.lookAt(this.scene.position)
  }
  initScene(){
    this.scene = new THREE.Scene()
  }
  initModel(){
    let geometry = new THREE.Geometry()
    geometry.vertices.push(new THREE.Vector3(-window.innerWidth, 0, 0))
    geometry.vertices.push(new THREE.Vector3(window.innerWidth, 0, 0))


    for(let i=0; i<20; i++){
      let material = new THREE.LineDashedMaterial({
        scale: 1,
        dashSize: 30,
        gapSize: 10,
      })


      material.color.setRGB(Math.random(), Math.random(), Math.random())
      let line = new THREE.Line(geometry, material)
      line.computeLineDistances()
      line.position.y = i*(window.innerHeight/20) - window.innerHeight/2
      this.scene.add(line)
    }

    for(let i=0; i<=20; i++){
      let material = new THREE.LineBasicMaterial({color: 0xffffff, opacity: 1})
      material.color.setRGB(Math.random(), Math.random(), Math.random())
      let line = new THREE.Line(geometry, material)
      line.position.x = i*(window.innerWidth/20) - window.innerWidth/2
      line.rotation.z = 90 * Math.PI/180
      this.scene.add(line)
    }

  }
  webGLRender(){
    this.renderer.render(this.scene, this.camera)
  }
  animate(){
    // this.camera.rotation.z = Date.now() * 0.001 * 0.1
    requestAnimationFrame(this.animate.bind(this))
    this.webGLRender()
  }
  run(){
    this.initScene()
    this.initCamera()
    this.initRenderer()
    this.initModel()
    this.animate()
  }
}

export default ThreeDrawBlockLine

import React from 'react'
import * as THREE from 'three'

class ThreeDrawLinkLine extends React.Component{
  constructor(props){
    super(props)
    this.renderer = null
    this.camera = null
    this.scene = null
    this.geometry = null
    this.vector = {
      x: Math.random(),
      y: Math.random(),
      z: 0
    }
    this.vectorDatas = []
    this.positions = new Float32Array(3000)
  }

  render(){
    return(
      <div ref="canvasBox" className="canvas-box"></div>
    )
  }

  componentDidMount() { this.run() }

  initRenderer(){
    this.renderer = new THREE.WebGLRenderer()
    this.renderer.setSize(window.innerWidth, window.innerHeight)
    this.refs.canvasBox.appendChild(this.renderer.domElement)
  }

  initCamera(){
    this.camera = new THREE.PerspectiveCamera(55, window.innerWidth/window.innerHeight, 2, 2000)
    this.camera.position.z = 500
    this.camera.lookAt(this.scene.position)
  }

  initScene(){
    this.scene = new THREE.Scene()
    this.scene.fog = new THREE.FogExp2(0x000000, 0.002)
  }

  initModel(){
    this.geometry = new THREE.BufferGeometry()

    for ( var i = 0; i < 1000; i ++ ) {
      this.positions[ i * 3 ] = Math.random() * 1000 - 500
      this.positions[ i * 3 + 1 ] = Math.random() * 1000 - 500
      this.positions[ i * 3 + 2 ] = Math.random() * 1000 - 500

      this.vectorDatas.push({
        velocity: new THREE.Vector3( - 1 + Math.random() * 2, - 1 + Math.random() * 2, - 1 + Math.random() * 2 )
      })      
    }

    this.geometry.setDrawRange( 0, 500 )

    this.geometry.addAttribute( 'position', new THREE.BufferAttribute( this.positions, 3 ) )
    let material = new THREE.PointsMaterial( { size: 3, sizeAttenuation: false, alphaTest: 0.5, transparent: true } )
    material.color.setRGB(Math.random(), Math.random(), Math.random())
    let point = new THREE.Line(this.geometry, material)
    this.scene.add(point)
  }

  webglRender(){ 
    this.renderer.render(this.scene, this.camera)
  }

  animate(){
    for(let i=0; i<1000; i++){
      if(this.positions[i * 3] >= 500 || this.positions[i * 3] <= 0){
        this.vectorDatas[i].velocity.x *= -1
      }
      if(this.positions[i * 3 + 1] >= 500 || this.positions[i * 3 + 1] <= 0){
        this.vectorDatas[i].velocity.y *= -1
      }
      if(this.positions[i * 3 + 2] >= 500 || this.positions[i * 3 + 2] <= 0){
        this.vectorDatas[i].velocity.z *= -1
      }

      this.positions[ i * 3 ] += this.vectorDatas[i].velocity.x
      this.positions[ i * 3 + 1 ] += this.vectorDatas[i].velocity.y
      this.positions[ i * 3 + 2 ] += this.vectorDatas[i].velocity.z
    }

    this.geometry.attributes.position.needsUpdate = true;
    this.camera.rotation.z = Date.now() * 0.001 * 0.1
    requestAnimationFrame(this.animate.bind(this))
    this.webglRender()
  }

  run(){
    this.initScene()
    this.initCamera()
    this.initRenderer()
    this.initModel()
    this.animate()
  }
}

export default ThreeDrawLinkLine
